#PBS -S /bin/bash
#PBS -lnodes=1:cores16
#PBS -lwalltime=24:00:00
cd "$TMPDIR"
cp  %(e_path)s train.e
cp  %(f_path)s train.f
cp %(ibm1_table_path)s ibm1.table
cp %(e_vocab_path)s train.e.vcb
cp %(f_vocab_path)s train.f.vcb
# call the programs
module load python
source %(script_dir)s/venv/bin/activate
python %(script_dir)s/word_alignment/models/zero_tree/get_parameters.py -e train.e -f train.f -group_size %(group_size)s -output_prefix %(job_name)s -t_file ibm1.table -e_voc train.e.vcb -f_voc train.f.vcb
gzip *.prms.u
cp %(job_name)s* %(it_dir)s
wait          # wait until programs are finished
