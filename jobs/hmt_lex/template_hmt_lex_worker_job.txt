#PBS -S /bin/bash
#PBS -lnodes=1:cores16
#PBS -lwalltime=24:00:00
part=$PBS_ARRAYID
cd "$TMPDIR"
cp %(dir)s/it0/%(job_name)s.corpus.${part}.gz corpus.psnt.gz
cp %(prev_it_dir)s/%(job_name)s.params.${part}.gz params.txt.gz
gunzip params.txt.gz
gunzip corpus.psnt.gz
# call the programs
module load python
source %(script_dir)s/venv/bin/activate
python %(script_dir)s/word_alignment/models/hmt_lex/hmt_lex_worker.py -corpus corpus.psnt -params params.txt -num_workers %(num_workers)s -p_0 %(p_0)s -alpha %(alpha)s -buffer_size %(buffer_size)s
gzip params.txt.counts
cp params.txt.counts.gz %(it_dir)s/%(job_name)s.counts.${part}.gz
wait          # wait until programs are finished
