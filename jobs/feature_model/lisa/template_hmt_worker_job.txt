#PBS -S /bin/bash
#PBS -lnodes=1:cores16
#PBS -lwalltime=%(wall_time_train)s
part=$PBS_ARRAYID
cd "$TMPDIR"

cp %(it0_dir)s/%(job_name)s.corpus.${part}.gz corpus.psnt.gz
cp %(prev_it_dir)s/%(job_name)s.params.${part}.gz params.txt.gz
cp %(prev_it_dir)s/%(job_name)s.convoc.${part}.gz convoc.txt.gz
cp %(it0_dir)s/%(job_name)s.vecs vecs
cp %(prev_it_dir)s/%(job_name)s.weights weights
gunzip params.txt.gz
gunzip corpus.psnt.gz
gunzip convoc.txt.gz

module load python
source %(script_dir)s/venv/bin/activate

python %(script_dir)s/word_alignment/models/feature_model/feature_model_worker.py -corpus corpus.psnt -params params.txt -vecs vecs -weights weights -convoc convoc.txt -num_workers %(num_workers)s -p_0 %(p_0)s
gzip params.txt.counts
cp params.txt.counts.gz %(it_dir)s/%(job_name)s.counts.${part}.gz
wait
