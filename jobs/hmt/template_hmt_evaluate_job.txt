#PBS -S /bin/bash
#PBS -lnodes=1:cores16

# call the programs
cd "$TMPDIR"
cp %(dir)s/it0/%(job_name)s.corpus.1.gz corpus.test.gz
cp %(it_dir)s/%(job_name)s.params.1.gz params.txt.gz
gunzip params.txt.gz

cp corpus.aligned %(it_dir)s/%(job_name)s.aligned.1
wait          # wait until programs are finished
