#PBS -S /bin/bash
#PBS -lnodes=1:cores16
#PBS -lwalltime=24:00:00
cd "$TMPDIR"
cp  %(snt)s ef.snt
cp  %(f_raw)s f.raw
cp %(gold_file)s gold.txt
# call the programs
module load python
source %(script_dir)s/venv/bin/activate
python %(script_dir)s/word_alignment/parsing/parse_corpus.py -snt ef.snt -raw_f f.raw
%(filter_gold)spython %(script_dir)s/word_alignment/utils/filter_gold.py ef.snt.filtered gold.txt
cp ef.snt.parsed %(dir)s/
cp ef.snt.filtered %(dir)s/
cp gold.txt.filtered %(dir)s/
wait          # wait until programs are finished
